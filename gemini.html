import express from "express"; import fetch from "node-fetch"; import cors from "cors"; const app = express(); app.use(cors()); app.use(express.json()); app.post("/chat", async (req, res) => { const { prompt } = req.body; try { const ollamaRes = await fetch("http://localhost:11434/api/generate", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ model: "phi3", prompt, stream: false }), }); const json = await ollamaRes.json(); console.log("OLLAMA REPLY:", json); // log debug res.json({ reply: json.response || "⚠️ Tidak ada jawaban dari AI." }); } catch (err) { console.error(err); res.json({ reply: "⚠️ Error server: " + err.message }); } }); app.listen(3000, () => console.log("✅ Server AI jalan di http://localhost:3000"));
